"use client";
import { useState } from "react";

export default function TheBeastLive() {
  const [input, setInput] = useState("");
  const [messages, setMessages] = useState([
    { role: "ai", text: "أهلاً بيك يا أمجد.. 'الوحش' صحي وجاهز يفلترلك سوق مدينتي والتجمع. أؤمر يا قناص!" }
  ]);
  const [loading, setLoading] = useState(false);

  // 🎯 ارزع المفتاح بتاعك هنا بين العلامتين "" ومحدش هيشوفه غيرك في الكود
  const apiKey = "حط_هنا_المفتاح_بتاعك"; 
  
  const systemPrompt = `أنت 'الوحش'، مستشار عقاري خبير وصايع في مدينتي والقاهرة الجديدة. 
تحدث باللهجة العامية المصرية القوية. ممنوع تقول 'تحت أمرك يا فندم'. 
وظيفتك: فلترة العملاء VVIP، حساب الـ ROI، وكتابة سكريبتات مبيعات قوية لأمجد.`;

  const handleSend = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!input.trim() || apiKey === "حط_هنا_المفتاح_بتاعك") return;

    const userMsg = input.trim();
    setMessages(prev => [...prev, { role: "user", text: userMsg }]);
    setInput("");
    setLoading(true);

    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          contents: [{ parts: [{ text: userMsg }] }],
          systemInstruction: { parts: [{ text: systemPrompt }] }
        })
      });
      const data = await response.json();
      const reply = data.candidates?.[0]?.content?.parts?.[0]?.text || "في ضغط على السيرفر، جرب تاني يا وحش.";
      setMessages(prev => [...prev, { role: "ai", text: reply }]);
    } catch (err) {
      setMessages(prev => [...prev, { role: "ai", text: "المفتاح فيه مشكلة أو النت فصل!" }]);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-[#05070e] text-white flex flex-col font-sans" dir="rtl">
      {/* Header الصايع */}
      <header className="border-b border-yellow-600/30 bg-[#030408] p-5 flex justify-between items-center shadow-lg">
        <div className="flex items-center gap-4">
          <div className="w-10 h-10 bg-yellow-600 rounded-lg flex items-center justify-center text-black font-bold text-xl shadow-[0_0_20px_rgba(202,138,4,0.4)]">D</div>
          <div>
            <h1 className="text-2xl font-bold text-white tracking-widest font-serif">DOMAIN HUB</h1>
            <p className="text-[10px] text-yellow-600 font-bold tracking-[0.2em] uppercase">The Beast Brain</p>
          </div>
        </div>
        <div className="flex items-center gap-2 bg-green-500/10 px-3 py-1 rounded-full border border-green-500/20">
          <span className="w-2 h-2 bg-green-500 rounded-full animate-pulse"></span>
          <span className="text-[10px] text-green-400 font-bold uppercase">System Active</span>
        </div>
      </header>

      {/* Chat Area */}
      <main className="flex-1 overflow-y-auto p-6 space-y-6 max-w-4xl mx-auto w-full">
        {messages.map((msg, i) => (
          <div key={i} className={`flex ${msg.role === "user" ? "justify-start" : "justify-end"}`}>
            <div className={`p-4 rounded-2xl max-w-[85%] text-sm leading-relaxed shadow-xl ${
              msg.role === "user" 
              ? "bg-gray-800 border border-gray-700 text-white rounded-tr-none" 
              : "bg-yellow-600/10 border border-yellow-600/30 text-gray-200 rounded-tl-none"
            }`}>
              {msg.text}
            </div>
          </div>
        ))}
        {loading && <div className="text-yellow-600 text-xs font-bold animate-pulse text-center">الوحش بيحلل الداتا... 🔍</div>}
      </main>

      {/* Input Field */}
      <footer className="p-6 bg-[#030408] border-t border-gray-800/50">
        <form onSubmit={handleSend} className="max-w-4xl mx-auto flex gap-3">
          <input 
            type="text" 
            value={input}
            onChange={(e) => setInput(e.target.value)}
            placeholder="أؤمر الوحش يا قناص..." 
            className="flex-1 bg-[#0f1322] border border-gray-700 rounded-2xl px-5 py-4 text-white outline-none focus:border-yellow-600 transition shadow-inner"
            disabled={loading}
          />
          <button 
            type="submit" 
            disabled={loading || !input.trim()}
            className="bg-yellow-600 text-black px-8 rounded-2xl font-bold hover:bg-yellow-500 transition active:scale-95 shadow-lg disabled:opacity-50"
          >
            إرسال
          </button>
        </form>
      </footer>
    </div>
  );
}
